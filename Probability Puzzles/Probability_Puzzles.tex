\begin{question}
Two gamblers A and B have $n+1$ and $n$ fair coins respectively. 
Find $P(A > B)$.
\end{question}

\begin{solution}
Define events:
\begin{itemize}
    \item $E$: A has more heads in first $n$ tosses
    \item $F$: B has more heads in first $n$ tosses
    \item $G$: Equal heads in first $n$ tosses
    \item $H$: A gets head in $(n+1)$th toss
\end{itemize}

By symmetry:
\[
P(E) = P(F) = x, \quad P(G) = y, \quad P(H) = \tfrac{1}{2}
\]

\[
2x + y = 1
\]

Required:
\[
P(A > B) = P(E) + P(G \cap H) = x + \tfrac{y}{2} = \tfrac{1}{2}
\]
\end{solution}
